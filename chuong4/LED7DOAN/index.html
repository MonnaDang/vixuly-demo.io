    3. Sơ đồ nguyên lý
    Code
    #INCLUDE <TV_16F887.C>
SIGNED INT8 I;

VOID MAIN(){
  SET_TRIS_D(0X00);
  SET_TRIS_B(0X00);
  
  WHILE(TRUE){
     FOR(I =0;I<100;I++){
        OUTPUT_D(MA7DOAN[I%10]);
        IF((I/10)==0){OUTPUT_B(0XFF);}
        ELSE OUTPUT_B(MA7DOAN[I/10]);
        DELAY_MS(80);
     }
     FOR(I = 99;I>0;I--){
        OUTPUT_D(MA7DOAN[I%10]);
        IF((I/10)==0){OUTPUT_B(0XFF);}
        ELSE OUTPUT_B(MA7DOAN[I/10]);
        DELAY_MS(80);
     }
  }
}

  3. ĐẾM TỪ 0-99 TRÊN 2 LED 7 ĐOẠN BẰNG PHƯƠNG PHÁP QUÉT LED:
    4. Sơ đồ nguyên lý
    Code
    #INCLUDE <TV_16F887.C>
UNSIGNED INT8 DEM,I,CH,DV;

VOID GM_7DOAN(UNSIGNED INT X){
  DV = MA7DOAN[X%10];
  CH = MA7DOAN[X/10];
  IF(CH==0XC0)CH=0XFF;
}

VOID HT_7DOAN(){
  FOR(I=0;I<50;I++){
     OUTPUT_D(DV);
     OUTPUT_LOW(PIN_E0);
     DELAY_MS(1);
     OUTPUT_HIGH(PIN_E0);
     
     OUTPUT_D(CH);
     OUTPUT_LOW(PIN_E1);
     DELAY_MS(1);
     OUTPUT_HIGH(PIN_E1);
  }
}

VOID MAIN(){
  SET_TRIS_D(0X00);
  SET_TRIS_E(0X00);
  
  WHILE(TRUE){
     FOR(DEM=0;DEM<100;DEM++){
        GM_7DOAN(DEM);
        HT_7DOAN();
     }
  }
}
