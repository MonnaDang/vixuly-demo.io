    #INCLUDE <16F887.H>
#FUSES HS               //SU DUNG DAO DONG NGOAI
#USE DELAY(CLOCK=20M)

UNSIGNED INT8 X=0;

VOID CHECK_BUTTON(){
  IF(!INPUT(PIN_E0)){  //KHIEM TRA NUT NHAN
     DELAY_MS(10);     //DELAY CHONG DOI
     IF(!INPUT(PIN_E0)){
        X=~X;          //DAO X
        WHILE(!INPUT(PIN_E0));     //CHO NHA PHIM
     }
  }
}

VOID MAIN(){
  SET_TRIS_D(0X00);
  SET_TRIS_E(0XFF);
  OUTPUT_D(X);
  WHILE(TRUE){
     CHECK_BUTTON();
     OUTPUT_D(X);      //XUAT X RA PORT D
  }
}

#USE DELAY(CLOCK=20M)

UNSIGNED INT8 X=0,DEM=0;

VOID CHECK_BUTTON(){
   IF(!INPUT(PIN_E0)){
       DELAY_MS(20);
       IF(!INPUT(PIN_E0)){
          DEM++;
          IF(DEM<=8)X = (X<<1)|0X01;
          ELSE X=X<<1; 
          IF(DEM==16)DEM=0;
          WHILE(!INPUT(PIN_E0));   //CHO NHA PHIM
       }
   }
}

VOID MAIN(){
  SET_TRIS_D(0X00);
  SET_TRIS_E(0XFF);
  OUTPUT_D(X);
  
  WHILE(TRUE){
     CHECK_BUTTON();
     OUTPUT_D(X);

  4. ĐẾM LÊN, XUỐNG HIỂN THỊ LED 7 ĐOẠN CHUYỂN TRẠNG THÁI BẰNG NÚT NHẤN
    5. Code
    #INCLUDE<TV_16F887.C>

SIGNED INT8 DEM=0;
UNSIGNED INT1 STATE = 1;         //LUC DAU DEM LEN

VOID DISPLAY(UNSIGNED INT8 NUM){
  OUTPUT_D(MA7DOAN[NUM%10]);
}

VOID CHECK_BUTTON(){
  IF(!INPUT(PIN_E0)){
     DELAY_MS(20);
     IF(!INPUT(PIN_E0)){
        STATE=~STATE;          //DAO TRANG THAI
        WHILE(!INPUT(PIN_E0));
     }
  }
}

VOID MAIN(){
  SET_TRIS_D(0X00);
  SET_TRIS_E(0XFF);
  OUTPUT_D(0);
  DEM=0;
  WHILE(TRUE){
     CHECK_BUTTON();
     IF(STATE==1){
        DEM++;
        IF(DEM > 9) DEM=0;
     }
     ELSE{
        DEM--;
        IF(DEM <0) DEM =9;
     }
     DISPLAY(DEM);
     DELAY_MS(100);
  }
}
