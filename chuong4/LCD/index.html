    #INCLUDE <TV_16F887.C>
#INCLUDE <TV_LCD.C>

UNSIGNED INT8 DEM;

VOID DISPLAY(UNSIGNED NUM){
  LCD_COMMAND(0XC0);                     //HIEN THI O VI TRI DAU TIEN CUA HANG 2
  LCD_DATA(NUM/10?NUM/10+0X30:0X20);
  LCD_DATA((NUM%10+0X30));
}

VOID MAIN(){
  SET_TRIS_D(0X00);
  SET_TRIS_E(0XFF);
  LCD_SETUP();
  LCD_COMMAND(0X80);                     //VI TRI DAU TIEN CUA HANG 1
  LCD_DATA("DEM TU 0-99:");
  
  WHILE(TRUE){
     FOR(DEM=0;DEM<100;DEM++){
        DISPLAY(DEM);
        DELAY_MS(100);
     }
  }
}

  2. ĐẾM LÊN XUỐNG HIỂN THỊ TRÊN LCD DÙNG 2 NÚT NHẤN START/PAUSE VÀ UP/DOWN:
    3. Sơ đồ nguyên lý
    Code
    #INCLUDE <TV_16F887.C>
#INCLUDE <TV_LCD.C>

SIGNED INT8 DEM;
UNSIGNED INT1 STATE,UP_DOWN;

VOID DISPLAY(UNSIGNED NUM){               //HAM HIEN THI LCD
  LCD_COMMAND(0XC0);                     //HIEN THI O VI TRI DAU TIEN CUA HANG THU 2
  LCD_DATA(NUM/10?NUM/10+0X30:0X20);
  LCD_DATA((NUM%10+0X30));
}

VOID CHECK_BUTTON(UNSIGNED INT8 PIN){     //HAM KIEM TRA NUT NHAN
  IF(!INPUT(PIN)){
     DELAY_MS(10);
     IF(!INPUT(PIN)){
        IF(PIN==PIN_C0)UP_DOWN=~UP_DOWN;
        IF(PIN==PIN_C1)STATE =~ STATE;
        WHILE(!INPUT(PIN));
     }
  }
}

VOID MAIN(){
  SET_TRIS_D(0X00);
  SET_TRIS_E(0X00);
  SET_TRIS_C(0XFF);
  LCD_SETUP();
  LCD_COMMAND(0X80);                     //VI TRI DAU TIEN CUA HANG 1
  LCD_DATA("DEM TU 0-99:");
  STATE = 0;
  UP_DOWN =0;
  DEM=0;
  WHILE(TRUE){
     CHECK_BUTTON(PIN_C1);               //KIEM TRA NUT NHAN START/PAUSE NEU NHAN THI CHO PHEM DEM 0(PAUSE) 1(START)
     IF(STATE==1){
        CHECK_BUTTON(PIN_C0);            //KIEM TRA NUT NHAN UP/DOW  0(UP) 1(DOWN)
        IF(UP_DOWN==0){
           DEM++;
           IF(DEM>99) DEM=0;
        }ELSE{
           DEM--;
           IF(DEM<0) DEM=99;
        }
        DISPLAY(DEM);
        DELAY_MS(100);
     }
  }
}

    4. Sơ đồ nguyên lý
    Code
    #INCLUDE <TV_16F887.C>
#INCLUDE <TV_LCD.C>

UNSIGNED INT8 GIO,PHUT,GIAY;

VOID DISPLAY(UNSIGNED NUM){
  LCD_DATA(NUM/10+0X30);
  LCD_DATA((NUM%10+0X30));
}

VOID MAIN(){
  SET_TRIS_D(0X00);
  SET_TRIS_E(0X00);
  LCD_SETUP();
  LCD_COMMAND(0X80);
  LCD_DATA("GIO PHUT GIAY");
  GIO=PHUT=GIAY=0;
  WHILE(TRUE){
     IF(GIAY == 59){
           GIAY =0;
           IF(PHUT ==59){
              PHUT = 0;
              IF(GIO == 24){
                 GIO = 0;
              }
              ELSE GIO++;
           }ELSE PHUT++;
        }
        ELSE GIAY++;
        LCD_COMMAND(0XC0);            //DIA CHI VI TRI HANG 2 
        DISPLAY(GIO);
        LCD_DATA("H ");
        DISPLAY(PHUT);
        LCD_DATA("P ");
        DISPLAY(GIAY);
        LCD_DATA("S ");
        DELAY_MS(10);
  }
}
