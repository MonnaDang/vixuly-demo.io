  1. BẬT TẮT 1 LED ĐƠN VỚI CHU KÌ DELAY LÀ 1S:
    2. Sơ đồ nguyên lý

    Code 
    #INCLUDE <16F887.H>        //khai bao thu vien
#USE DELAY(CLOCK=20M)      //khai bao gia tri cua tu thach anh

UNSIGNED INT1 X = 0;           //khai bao bien x 
  
VOID MAIN(){
  SET_TRIS_D(0XFE);       //khoi tao port d
  OUTPUT_D(X);
  WHILE (TRUE){
     X=~X;                //dao x;
     OUTPUT_D(X);         //xuat x ra port d
     DELAY_MS(1000);      //delay 1s
  }
}
    Output
    Giải thích code 

    #INCLUDE <16F887.H>        
    Khai báo thư viện cho vi xử lý;

    #USE DELAY(CLOCK=20M)      
    Khai báo tần số của tụ thạch anh để sử dụng các hàm DELAY;

    UNSIGNED INT1 X = 0;
    Khai báo biến x kiểu không dấu INT1 (0÷1)bit với giá trị là 0;

    SET_TRIS_D(0XFE);
    Khởi tạo port D (0 là ngõ ra,1 là ngõ vào), vì chỉ sử dụng 1 chân PIN_D0 -> 0b1111110;

    OUTPUT_D(X);
    Xuất x ra port D (x = 0 hay 0b00000000) làm cho led tắt;

    WHILE (TRUE)
    Biểu thức điều kiện là TRUE nên các lệnh trong WHILE sẽ lặp lại vô hạn

    X=~X;
    Đảo x, giá trị của x lúc đầu là 0 đảo thành 1, sau đó thực hiện cách lệnh dưới rồi lặp lại do WHILE đảo lại 0, cứ thế lặp lại vô hạn;

    OUTPUT_D(X);
    Xuất x ra port D(x có giá trị là 1) làm cho led sáng;

    DELAY_MS(1000);
    Delay 1000ms, sau đó lặp lại các lệnh trong WHILE;

  2. SÁNG DẦN VỚI 8LED ĐƠN:
    3. Sơ đồ nguyên lý
    Code 
    #INCLUDE <16F887.H>        //khai bao thu vien
#USE DELAY(CLOCK=20M)      //khai bao gia tri cua tu thach anh

UNSIGNED INT8 X =0,I;      //khai bao bien x,I         
  
VOID MAIN(){                
  SET_TRIS_D(0X00);       //khoi tao port d la ngo ra
  OUTPUT_D(X);            //xuat x ra port d
  WHILE (TRUE){
     FOR(I = 0;I<8;I++){
        X=(X<<1)|1;       //dich trai 1bit roi or voi 1;
        OUTPUT_D(X);
        DELAY_MS(500);
     }
     FOR(I = 0;I<8;I++){
        X=(X<<1);         //dich trai 1 bit
        OUTPUT_D(X);
        DELAY_MS(500);
     }
  }
}
    Output
    Giải thích code 
    FOR(I = 0;I<8;I++){
    Lặp 8 lần, I tăng từ 0 tới 7;

    X=(X<<1)|1;
    Gán x bằng x dịch 1 rồi or với 1(hoặc cộng với 1) làm led sáng dần lên
    VD: X khởi tạo bằng 0(0b00000000)
    X=(X<<1)|1 => X= 0b00000001;
    X=(X<<1) => X = 0b00000010; sau đó X = X|1 => X = 0b00000011;
    ...
    Lặp lại cho tới khi biểu thức điều kiện của FOR sai; 
    OUTPUT_D(X);
    Xuất x ra port D (1 led sáng,0 led tắt); 

    X=(X<<1);
    Ở vòng lặp FOR thứ 2 dùng dịch bit để làm led tắt dần;
    Sử dụng IF thay cho FOR 
    #INCLUDE <16F887.H>        //khai bao thu vien
#USE DELAY(CLOCK=20M)      //khai bao gia tri cua tu thach anh

UNSIGNED INT8 X = 0;      //khai bao bien x        
SIGNED INT8 DEM = 0;      //khai bao bien Dem 
VOID MAIN(){                
  SET_TRIS_D(0X00);       //khoi tao port d la ngo ra
  OUTPUT_D(X);            //xuat x ra port d
  WHILE (TRUE){
     IF(DEM<8){
        X=(X<<1)|0X01;    //dich trai 1bit roi or voi 1;
     }
     ELSE{
        X=X<<1;           //dich trai 1 bit
     }
     OUTPUT_D(X);
     DELAY_MS(200);
     IF(DEM==15)DEM=-1;   
     DEM++;               
  }
}
  3. SÁNG DẦN VỚI 16LED ĐƠN
    4. Sơ đồ nguyên lý
    Code 
    #INCLUDE <16F887.H>        //khai bao thu vien
#USE DELAY(CLOCK=20M)      //khai bao gia tri cua tu thach anh

UNSIGNED INT16 X = 0;      //khai bao bien x        
SIGNED INT8 DEM = 0;      //khai bao bien Dem 
VOID MAIN(){                
  SET_TRIS_D(0X00);       //khoi tao port d
  SET_TRIS_C(0X00);
  OUTPUT_D(X);            //xuat x ra port d
  OUTPUT_C(X>>8);
  WHILE (TRUE){
     IF(DEM<16){
        X=(X<<1)|0X01;    //dich trai 1bit roi or voi 1;
     }
     ELSE{
        X=X<<1;           //dich trai 1 bit
     }
     OUTPUT_D(X);
     OUTPUT_C(X>>8);      //xuat 8 bit cao cua x
     DELAY_MS(200);
     IF(DEM==31)DEM=-1;   
     DEM++;               
  }
}
    Output
    Giải thích code 
    UNSIGNED INT16 X = 0;
    Khai báo biến X có kiểu dữ liệu 16 bit để xuất ra 16 led

    OUTPUT_D(X);
    OUTPUT_C(X>>8);
    Do X có kiểu 16bit nên khi xuất ra các port các bit tràn(bit cao) sẽ bị bỏ đi, chỉ xuất các bít thấp nên chỉ cần dùng OUTPUT_D(x) để xuất các bít thấp của x;
    Để xuất các bit cao ra port C thì ta dịch phải 8bit
    VD X=0b1111 11111 0000 0000;
    X=>>8 khi đó X = 0b0000 0000 11111 11111;
  4. SÁNG DỒN VỚI 16LED ĐƠN
    5. Code 
    #INCLUDE <16F887.H>        //khai bao thu vien
#USE DELAY(CLOCK=20M)      //khai bao gia tri cua tu thach anh

UNSIGNED INT16 X,Y,Z;      //khai bao bien x,y,z   
UNSIGNED INT8 I,J;

VOID SANG_LED(INT16 X){
  OUTPUT_D(X);
  OUTPUT_C(X>>8);
  DELAY_MS(200);
}

VOID MAIN(){                
  SET_TRIS_D(0X00);       //khoi tao port d
  SET_TRIS_C(0X00);       //khoi tao port c
  WHILE (TRUE){
     X=Y=0;
     FOR(I=0;I<16;I++){
        Z=1;
        FOR(J=I;J<16;J++)
        {
           Y=Z|X;
           SANG_LED(Y);
           Z=Z<<1;
        }
        X=Y;
     }  
  }
}
    Output
    Giải thích code 
    VOID SANG_LED(INT16 X){
  OUTPUT_D(Y);
  OUTPUT_C(Y>>8);
  DELAY_MS(200);
}

    Hàm con SANG_LED cần đưa vào 1 tham số x có kiểu dữ liệu là 16bit để xuất ra 2 port D và C;


    X=Y=0;
FOR(I=0;I<16;I++){
  Z=1;
  FOR(J=I;J<16;J++)
    {
       Y=Z|X;
       SANG_LED(Y);
       Z=Z<<1;
    }
    X=Y;
 }  

    Trong vòng lặp while ta có:
    Đầu tiên x = y = 0;
    I = 0:
      Z=1;
      J=I=0<16 đúng;
        Y=Z|X; //Y = 0b0000 0000 0000 0001 (0x0001)
        Sáng led 1;
        Z=Z<<1; //Z= 0x0002;
        J+1;     J=1<16 đúng
        Quay lại thực hiện các lệnh trong vòng lặp J;
        Y=0x0002;
        Sáng led 2;
        Z=Z<<1; //Z=0x0004;
        Tiếp tục tăng J rồi thực hiện sáng hết 16 led;
      Sau đó gán biến Y cho Z; //Z = 0x8000;
    Tiếp tục tăng I và thực hiện vòng lặp cho đến khi biểu thức điều kiện trong For sai thì lặp lại vòng lặp While bắt đầu 1 chu kì mới;
